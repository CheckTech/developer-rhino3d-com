```vbnet
Sub SelText
  Dim arrObjects, strObject
  arrObjects = Rhino.AllObjects
  If IsArray(arrObjects) Then
    For Each strObject In arrObjects
      If Rhino.IsText(strObject) Then
        Rhino.SelectObject strObject
      End If
    Next
  End If
End Sub
```
